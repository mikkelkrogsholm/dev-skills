src/
├── index.ts                      # Bun entry: builds Hono app, calls .fetch, starts server
├── app.ts                        # Composes root Hono app: global middleware + mounts feature routers
├── env.ts                        # Zod-validated process.env; throws at boot on invalid config
│
├── config/
│   ├── constants.ts              # App-wide constants (pagination limits, timeouts, role names)
│   └── logger.ts                 # Structured logger (pino); one instance, imported everywhere
│
├── db/
│   ├── client.ts                 # Neon serverless driver + drizzle() instance (singleton export `db`)
│   ├── schema/
│   │   ├── index.ts              # Re-exports every table; Drizzle needs a single schema object
│   │   ├── users.ts              # users, sessions tables (owned by auth/users features)
│   │   ├── billing.ts            # customers, subscriptions, invoices tables
│   │   └── _shared.ts            # Common column helpers: id(), timestamps(), softDelete()
│   ├── migrations/               # Generated by drizzle-kit; never hand-edit
│   │   └── meta/
│   └── seed.ts                   # Idempotent seed script, run via `bun run db:seed`
│
├── features/                     # Vertical slices. One folder = one bounded context.
│   │
│   ├── auth/
│   │   ├── auth.routes.ts        # Hono sub-app: POST /login, /logout, /refresh, /register
│   │   ├── auth.service.ts       # Pure business logic: hashPassword, verifySession, issueToken
│   │   ├── auth.repo.ts          # DB access for sessions (uses db + schema/users)
│   │   ├── auth.schemas.ts       # Zod: LoginInput, RegisterInput, SessionDTO
│   │   ├── auth.middleware.ts    # requireAuth, requireRole — used by other features
│   │   └── auth.errors.ts        # InvalidCredentialsError, SessionExpiredError
│   │
│   ├── users/
│   │   ├── users.routes.ts       # GET /users/me, PATCH /users/me, GET /users/:id (admin)
│   │   ├── users.service.ts      # createUser, updateProfile, deactivateUser
│   │   ├── users.repo.ts         # All Drizzle queries for users table
│   │   ├── users.schemas.ts      # UserDTO, UpdateUserInput, ListUsersQuery
│   │   └── users.errors.ts       # UserNotFoundError, EmailTakenError
│   │
│   └── billing/
│       ├── billing.routes.ts     # GET /billing/subscription, POST /billing/checkout, webhooks
│       ├── billing.service.ts    # startCheckout, applyWebhookEvent, cancelSubscription
│       ├── billing.repo.ts       # subscriptions + invoices queries
│       ├── billing.schemas.ts    # CheckoutInput, WebhookPayload, SubscriptionDTO
│       ├── billing.webhooks.ts   # Provider-signed webhook verification + dispatch
│       └── billing.errors.ts     # NoActiveSubscriptionError, WebhookSignatureError
│
├── middleware/                   # Cross-cutting only. Feature-specific middleware lives in the feature.
│   ├── request-id.ts             # Attach/propagate X-Request-Id
│   ├── logger.ts                 # Per-request structured log line
│   ├── error-handler.ts          # Converts thrown AppError → JSON response; unknown → 500
│   ├── cors.ts                   # CORS config wrapper
│   └── rate-limit.ts             # Upstash/in-memory limiter
│
├── lib/                          # Framework-agnostic utilities. No Hono, no DB imports.
│   ├── errors.ts                 # AppError base class + HTTP status mapping
│   ├── result.ts                 # Result<T, E> helpers (optional pattern)
│   ├── crypto.ts                 # hash, verify, randomToken (wraps Bun.password / Web Crypto)
│   ├── id.ts                     # ULID/UUID generation
│   └── date.ts                   # Time helpers; always UTC
│
├── http/
│   ├── response.ts               # ok(), created(), noContent() — uniform response shape
│   ├── pagination.ts             # parseCursor, buildPageMeta
│   └── openapi.ts                # Optional: @hono/zod-openapi registry if used
│
└── types/
    ├── env.d.ts                  # Ambient types for Bun.env
    └── hono.d.ts                 # Hono Variables/Bindings augmentation (e.g. c.get('userId'))

tests/                            # Mirrors src/ one-to-one
├── features/
│   ├── auth/auth.service.test.ts
│   ├── users/users.repo.test.ts
│   └── billing/billing.webhooks.test.ts
├── integration/                  # Route-level: spin up app.fetch, hit real test DB
│   └── auth.routes.test.ts
└── helpers/
    ├── test-db.ts                # Creates throwaway Neon branch per test run
    └── factories.ts              # makeUser(), makeSubscription() builders

drizzle.config.ts                 # Points to src/db/schema + migrations dir
package.json
tsconfig.json
bunfig.toml
